### 9461. 파도반 수열

(주소)https://www.acmicpc.net/problem/9461


#### 문제 요약:


#### 풀이 해설:


```c++
#include <iostream>
#include <algorithm>
#include <queue>
using namespace std;

int T, n;
long long dp[102];

int main() {
    
    cin>>T;
    dp[0]=0;
    dp[1]=1;
    dp[2]=1;
    dp[3]=1;
    
    for(int i=0;i<T;i++){
        cin>>n;
        if(n<4) cout<<dp[n]<<"\n";
        else{
            for(int j=4;j<=n;j++){
                dp[j]=dp[j-3]+dp[j-2];    //점화식
            }
            cout<<dp[n]<<"\n";
        }
    }
}
```

### 2146. 다리 만들기

(주소)https://www.acmicpc.net/problem/2146


#### 문제 요약:


#### 풀이 해설:

섬 나타날때는 dfs, 각 섬마다 확장해나가면서 가장 최소 길이의 다리를 구할때는 bfs 이용.

```c++
#include <iostream>
#include <algorithm>
#include <queue>
using namespace std;

int N, num=1, result=0, rr=2147483647;
int map[101][101];
int dir[4][2]={{0,1},{0,-1},{1,0},{-1,0}};
bool visited[101][101];
queue< pair<int,int> > q;

void dfs(int x, int y, int num){
    
    for(int i=0;i<4;i++){
        int dx=x+dir[i][0];
        int dy=y+dir[i][1];
        if(0<=dx && dx<N && 0<=dy && dy<N){
            if(visited[dx][dy]==false && map[dx][dy]==1){
                visited[dx][dy]=true;
                map[dx][dy]=num;
                dfs(dx,dy,num);
            }
        }
    }
}

int bfs(int num){
    result=0;
    
    for(int i=0;i<N;i++){
        for(int j=0;j<N;j++){
            if(map[i][j]==num) {
                visited[i][j]=true;
                q.push(make_pair(i,j));
            }
        }
    }
    
    while(!q.empty()){
        int s = q.size();
        
        while(s--){  
            int x = q.front().first;
            int y = q.front().second;
            q.pop();
            for(int j=0;j<4;j++){
                int dx=x+dir[j][0];
                int dy=y+dir[j][1];
                
                if(0<=dx && dx<N && 0<=dy && dy<N){
                    
                    if(visited[dx][dy]==false && map[dx][dy]==0){
                        q.push(make_pair(dx,dy));
                        visited[dx][dy]=true;    //바다를 지날때는 queue에 위치값넣기
                    }
                    else if(map[dx][dy]!=0 && map[dx][dy]!=num) return result;
                                                //다른 번호의 섬을 만나면 result 리턴
                }
            }
            
        }
        result++;
    }
    return result;
}

int main() {
    cin>>N;
    
    for(int i=0;i<N;i++){
        for(int j=0;j<N;j++){
            cin>>map[i][j];
        }
    }
    for(int i=0;i<N;i++){
        for(int j=0;j<N;j++){
        	if(visited[i][j]==false && map[i][j]!=0){
        		visited[i][j]=true;
	            map[i][j]=num;
	            dfs(i,j, num);
	            num++;
        	}
            
        }
    }     //dfs로 섬 탐색 끝 ( 섬마다 번호 지정 ) 
    
    for(int i=1;i<num;i++){
        for(int k=0;k<N;k++){
            for(int j=0;j<N;j++){
                visited[k][j]=false;   //초기화
            }
        }
        for(int g=0;!q.empty();g++){   //초기화
        	q.pop();
        }
        
        rr=min(rr,bfs(i));
    }
    cout<<rr;
    
}
```
