### 14888. 연산자 끼워넣기
(주소)https://www.acmicpc.net/problem/14888


#### 풀이 해설:

수정필요 


```c++
#include<iostream>
#include<algorithm>
#include<vector>
#include<climits>

using namespace std;

int N;
int number[11], cal[4];
int mx=INT_MIN, mn=INT_MAX;
bool visited[10]; // 연산자 방문 여부
vector<char> v;

//cnt: 연산자 방문 개수, result: 계산 결과, idx: 인덱스

void dfs(int idx, int cnt, int result){
    if(cnt==N-1){
        if(mx<result) mx=result;
        if(mn>result) mn=result;
    }
    else{
        for(int i=0;i<N-1;i++){
            if(visited[i]==false){
                visited[i]=true;
                switch(v[i]){
                    case '+':
                        result = result + number[idx];
                        break;
                    case '-':
                        result = result - number[idx];
                        break;
                    case '*':
                        result = result * number[idx];
                        break;
                    case '/':
                        result = result / number[idx];
                        break;
                }
                dfs(idx+1,cnt+1,result);
                visited[i]=false;
            }
        }
    }
}

int main(int argc, char** argv)
{
	cin>>N;
    for(int i=0;i<N;i++){
        cin>>number[i];
    }
    for(int i=0;i<4;i++){
        cin>>cal[i];
        for(int j=0;j<cal[i];j++){
            if(i==0) v.push_back('+');
            if(i==1) v.push_back('-');
            if(i==2) v.push_back('*');
            if(i==3) v.push_back('/');
        }
    }
    dfs(1, 0, number[0]);  // idx, cnt, result
    cout<<mx<<"\n";
    cout<<mn<<"\n";
}
```

### 14889. 스타트와 링크
(주소)https://www.acmicpc.net/problem/14889


#### 풀이 해설:

조합 함수 2가지 -> make_team, calculate

```c++
#include <iostream>
#include <algorithm>
#include <map>
#include <vector>
#include <climits>
#include <stdlib.h>
using namespace std;

int N;
int sum=0, sum1=0, sum2=0;
int answer=INT_MAX;
int S[20][20];
bool visited[21];
bool visited2[21];
vector<int> v;
vector<int> remain;
map<vector<int>, int> mp;
vector<int> v2;

void calculate(vector<int>v, int x, int cnt){
	
    if(cnt==2){
        //계산
        sum+=S[v2[0]-1][v2[1]-1];
        sum+=S[v2[1]-1][v2[0]-1];
        return;
    }
    for(int i=x;i<N/2;i++){
        if(visited[i]==false){
            visited[i]=true;
            v2.push_back(v[i]);
            calculate(v, i+1, cnt+1);
            v2.pop_back();
            visited[i]=false;
        }
    }
}

void make_team(int x, int cnt){
    if(cnt==N/2){  // ex. v: 1,2,3 / remain:4,5,6
        if(mp[v]==0){  // 이유 - 계산 중복방지: (1,2,3/4,5,6) == (4,5,6/1,2,3)
            int a = 0;
            for(int i=0;i<N;i++){
                a=0;
                for(int j=0;j<v.size();j++){
                    if(i+1 == v[j]) a=1;
                }
                if(a==0) remain.push_back(i+1);
            }

            //차이 구하기
            sum=0;
            for(int i=0;i<N/2;i++){
            	visited[i]=false;
            }
            calculate(v,0,0);
            sum1=sum;

            sum=0;
            for(int i=0;i<N/2;i++){
            	visited[i]=false;
            }
            calculate(remain,0,0);
            sum2=sum;
            
            answer=min(answer, abs(sum1-sum2));
            
            mp[v]++;
            mp[remain]++;
            remain.clear();
        }
        
        return;
    }
    for(int i=x;i<N;i++){
        if(visited[i+1]==false){
            visited[i+1]=true;
            v.push_back(i+1);
            make_team(i+1, cnt+1);
            v.pop_back();
            visited[i+1]=false;
        }
    }
}

int main(){
	cin>>N;
    for(int i=0;i<N;i++){
        for(int j=0;j<N;j++){
            cin>>S[i][j];
        }
    }
    make_team(0,0);
    cout<<answer;
}
```

### 14502. 연구소
(주소)https://www.acmicpc.net/problem/14502


#### 풀이 해설:

벽을 세울 수 있는 조합들을 모두 구해서 (dfs) 이에 대해 bfs 수행.

```c++
#include <iostream>
#include <algorithm>
#include <queue>
#include <vector>
using namespace std;

int N,M;
int num=0, nn=0;
int map[8][8];
int map2[8][8];
bool visited[8][8];
bool visited_dfs[64];
vector<pair<int,int>>zero;
vector<pair<int,int>>virus;
vector<pair<int,int>>wall;
queue<pair<int,int>>q;
int dir[4][2] = {{1,0},{0,-1},{-1,0},{0,1}};

void bfs(){
    for(int i=0;i<virus.size();i++){
        q.push(virus[i]);
        visited[virus[i].first][virus[i].second]=true;
    }
    while(!q.empty()){
        int x = q.front().first;
        int y = q.front().second;
        q.pop();
        for(int i=0;i<4;i++){
            int dx = x + dir[i][0];
            int dy = y + dir[i][1];
            if(0<=dx && dx<N && 0<=dy && dy<M){
                if(visited[dx][dy]==false && map2[dx][dy]==0){
                    visited[dx][dy]=true;
                    map2[dx][dy]=2;
                    q.push({dx,dy});
                }
            }
        }
    }
}

void dfs(int x, int cnt){
    if(cnt==3){
        for(int i=0;i<N;i++){
            for(int j=0;j<M;j++){
                map2[i][j]=map[i][j];  // 복사
                visited[i][j]=false;
            }
        }
        for(int i=0;i<wall.size();i++){
            map2[wall[i].first][wall[i].second]=1;  // 벽 세움
        }
        bfs();
        for(int i=0;i<N;i++){
            for(int j=0;j<M;j++){
                if(map2[i][j]==0) nn++;
            }
        }
        num=max(num,nn);
        nn=0;
        
        return;
    }
    for(int i=x;i<zero.size();i++){
        if(visited_dfs[i]==false){
            visited_dfs[i]=true;
            wall.push_back(zero[i]);
            dfs(i+1, cnt+1);
            wall.pop_back();
            visited_dfs[i]=false;
        }
    }
}

int main(){
	cin>>N>>M;
    
    for(int i=0;i<N;i++){
        for(int j=0;j<M;j++){
            cin>>map[i][j];
            if(map[i][j]==0) zero.push_back({i,j});
            if(map[i][j]==2) virus.push_back({i,j});
        }
    }
    
    dfs(0,0); // 벽 세울 3개 조합 구해서 bfs 실행.
    cout<<num;
    
}
```

### 2105. [모의 SW 역량테스트] 디저트 카페
(주소)https://swexpertacademy.com/main/code/problem/problemDetail.do?contestProbId=AV5VwAr6APYDFAWu


#### 풀이 해설:

idx-=4; 가 아니라 왜 break; 인지 모르겠음

```c++
#include<iostream>
#include<algorithm>
#include<map>

using namespace std;

int T, N;
int answer=0, sum=0, idx=0, cnt=0;
int startX, startY;
int D[20][20];
bool visited[20][20];
int dir[4][2] = {{1,1},{1,-1},{-1,-1},{-1,1}};  // 대각선 4방향
map<int, int> mp;
map<int,int>::iterator it;

void dfs(int x, int y, int idx){  // 백트래킹

    if(x==startX && y==startY && cnt>1){
        answer=max(answer, cnt);
        return;
    }
    for(int i=0;i<2;i++){  // 오던 방향에서 2방향만 더 가볼 수 있음 -> 그래야 사각형
        if(idx+i>3) break;   // idx-=4;
        int dx = x+dir[idx+i][0];
        int dy = y+dir[idx+i][1];

        if(0<=dx && dx<N && 0<=dy && dy<N){
            if((visited[dx][dy]==false && mp[D[dx][dy]]==0)){
                mp[D[dx][dy]]++;
                visited[dx][dy]=true;
                cnt++;
                
                dfs(dx,dy,idx+i);

                cnt--;
                visited[dx][dy]=false;
                mp[D[dx][dy]]--;
            }
            else if(dx==startX && dy==startY){
                dfs(dx,dy,idx+i);
            }
        }
    }
    
}

int main(int argc, char** argv)
{
	int test_case;
	
	cin>>T;
	
	for(test_case = 1; test_case <= T; ++test_case)
	{
        cin>>N;
        idx=0;
        answer=-1;
        cnt=0;
        mp.clear();
        
        for(int i=0;i<N;i++){
            for(int j=0;j<N;j++){
                cin>>D[i][j];
                visited[i][j]=false;
            }
        }
        
        for(int i=0;i<N;i++){
            for(int j=0;j<N;j++){
            	cnt=0;
                mp.clear();
                visited[i][j]=true;
                mp[D[i][j]]++;
                startX=i;
                startY=j;
                cnt++;
                dfs(i,j,0);          // 출발점 쌍, 시작 방향 인덱스
                visited[i][j]=false;
                mp[D[i][j]]--;
            }
        }
        
        cout<<"#"<<test_case<<" "<<answer<<"\n";
	}
	return 0;//정상종료시 반드시 0을 리턴해야합니다.
}
```
