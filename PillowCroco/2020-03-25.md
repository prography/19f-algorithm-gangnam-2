### 소용돌이 예쁘게 출력하기(골4)
```Python
import sys

r1, c1, r2, c2 = list(map(int, sys.stdin.readline().split()))

# 가장 큰 수의 자리수를 구함.
# 가장큰 수는 (r1, c1), (r2, c1), (r2, c2) 중에 생김
max_num = 0
for a in [r1, r2]:
    for b in [c1, c2]:
        i = max(abs(a), abs(b))
        n = 0
        if a == -i:
            n = (2 * i + 1) ** 2 - (2 * i) - (2 * i) - b - i

        if a == i:
            n = (2 * i + 1) ** 2 - (2 * i) + b + i

        if b == -i:
            n = (2 * i + 1) ** 2 - (2 * i) - (2 * i) + a + i

        if b == i:
            n = (2 * i + 1) ** 2 - (2 * i) - (2 * i) - (2 * i) - a - i

        max_num = max(max_num, n)

# 그 자리수 + 1만큼이 숫자간 간격이 되어야함.
tabsize = len(str(max_num)) + 1

ans = ""


# 이제 채워넣자. i번째 도넛의 끝 (i,i)엔 2i+1의 제곱이 들어감.
for a in range(r1, r2 + 1):
    for b in range(c1, c2 + 1):
        i = max(abs(a), abs(b))
        # if i == 0:
        #     n = 1
        #     print(" " * (tabsize - len(str(n))), n, end="")
        #     continue

        if a == -i:
            n = (2 * i + 1) ** 2 - (2 * i) - (2 * i) - b - i
            if b == c1:
                # # print(n, end="")
                ans += " " * (tabsize - len(str(n)) - 1)
                ans += str(n)
            else:
                # print(" " * (tabsize - len(str(n))), n, end="")
                ans += " " * (tabsize - len(str(n)))
                ans += str(n)
            continue

        if a == i:
            n = (2 * i + 1) ** 2 - (2 * i) + b + i
            if b == c1:
                # # print(n, end="")
                ans += " " * (tabsize - len(str(n)) - 1)
                ans += str(n)
            else:
                # print(" " * (tabsize - len(str(n))), n, end="")
                ans += " " * (tabsize - len(str(n)))
                ans += str(n)
            continue

        if b == -i:
            n = (2 * i + 1) ** 2 - (2 * i) - (2 * i) + a + i
            if b == c1:
                # # print(n, end="")
                ans += " " * (tabsize - len(str(n)) - 1)
                ans += str(n)
            else:
                # print(" " * (tabsize - len(str(n))), n, end="")
                ans += " " * (tabsize - len(str(n)))
                ans += str(n)
            continue

        if b == i:
            n = (2 * i + 1) ** 2 - (2 * i) - (2 * i) - (2 * i) - a - i
            if b == c1:
                # # print(n, end="")
                ans += " " * (tabsize - len(str(n)) - 1)
                ans += str(n)
            else:
                # print(" " * (tabsize - len(str(n))), n, end="")
                ans += " " * (tabsize - len(str(n)))
                ans += str(n)
            continue

    ans += "\n"

print(ans)
```

###아기 상어 뚜루룻뚜루(SWEA) (골5)

```Python
import sys

d = [[1, 0], [-1, 0], [0, 1], [0, -1]]

# bfs는 해본적이없구나..
def searchFish(x, y, size, memo, visited):
    visited[x][y] = True
    q = list()
    q.append((x, y, 0))

    while q:
        x, y, dist = q.pop(0)
        for dx, dy in d:
            nx, ny = x + dx, y + dy
            if 0 <= nx < N and 0 <= ny < N and visited[nx][ny] is False:
                if g[nx][ny] == 0 or g[nx][ny] == size:
                    visited[nx][ny] = True
                    q.append((nx, ny, dist + 1))

                elif g[nx][ny] < size:
                    visited[nx][ny] = True
                    memo.append([dist + 1, nx, ny])


def baby_shark(g):
    visited = [[False for _ in range(N)] for _ in range(N)]
    total_dist = 0
    memo = []
    size = 2
    remaining = size
    # 1st iter
    x, y = 0, 0
    for i, row in enumerate(g):
        for j, num in enumerate(row):
            if num is 9:
                x, y = i, j

    while True:
        searchFish(x, y, size, memo, visited)
        g[x][y] = 0

        # if no edible fish around, finish.
        if not memo:
            return total_dist

        # check through memo and go eat the nearest one.
        memo.sort(key=lambda n: (n[0], n[1], n[2]))

        dist, x, y = memo[0]
        total_dist += dist
        remaining -= 1
        # print(memo)
        # print("[%d, %d], dist=%d, size=%d, remaining=%d, total=%d"
        #       %(memo[0][1],memo[0][2],memo[0][0], size, remaining, total_dist))
        # print("------------------")
        memo.clear()
        visited = [[False for _ in range(N)] for _ in range(N)]

        if not remaining:
            size += 1
            remaining = size


if __name__ == "__main__":
    N = int(input())
    g = []
    for _ in range(N):
        g.append(list(map(int, sys.stdin.readline().split())))

    print(baby_shark(g))
```
