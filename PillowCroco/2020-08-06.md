### 최대 유량
https://www.acmicpc.net/problem/6086

visited를 안쓰고 prev를 쓴 것이 인상적!

`for(int i=25; i!=0; i=prev[i])` 문이 아주 좋았어요

```c++
#include<iostream>
#include<vector>
#include<algorithm>
#include<queue>
using namespace std;

int c[52][52]{ 0 };
int f[52][52]{ 0 };

inline int convert(char c) {
	if (c <= 'Z') return c - 'A';
	return c - 'a' + 26;
}

int main() {
	int n;
	char s, d;
	int w;


	vector<int> adj[52];

	cin >> n;
	while (n--) {
		cin >> s >> d >> w;
		s = convert(s);
		d = convert(d);
		c[s][d] = c[d][s] += w;

		adj[s].push_back(d);
		adj[d].push_back(s);
	}

	int total = 0;
	while (true) {
		//visited 대신에 previous idx를 저장해놓는 걸로 대체!
		int prev[52];
		fill(prev, prev+52, -1);
		//#include<string> 하면 memset(prev, -1, 52*sizeof(int))가능
		queue<int> Q;

		//bfs를 통해 A->Z까지 가는 모든 길들을, prev를 기록하면서 찾아냄.
		//그 길들 중 c[curr][next] - f[curr][next] > 0를 통해
		//여유 유량이 있을때만 찾는 것을 확인!
		Q.push(0);
		while (!Q.empty() && prev[25] == -1) {
			int curr = Q.front();
			Q.pop();
			for (int next : adj[curr]) {
				if (c[curr][next] - f[curr][next] > 0 && prev[next] == -1) {
					Q.push(next);
					prev[next] = curr;
					if (next == 25)
						break;
				}
			}
		}

		if (prev[25] == -1)
			break;

		int eff_flow = 1234567890;

		//prev 거슬러 올라가기! c-f가 가장 최소인 애를 기준으로 맞춰준다.
		for (int i = 25; i != 0; i = prev[i]) {
			eff_flow = min(eff_flow, c[prev[i]][i] - f[prev[i]][i]);
		}

		for (int i = 25; i != 0; i = prev[i]) {
			f[prev[i]][i] += eff_flow;
			f[i][prev[i]] -= eff_flow;
		}

		total += eff_flow;

	}

	cout << total << endl;
	return 0;
}
```

### Reorganize String
https://leetcode.com/problems/reorganize-string/

EZ ..

아니 pair만들때 `make_pair(first,second)` 이렇게 할필요 없이

`{first, second}` 이래 해도 되는거였네 개멍청했다 `make_pair`는 그럼 왜필요한거지

```c++
// map에 넣고 pq에서 많이 있는 순서대로 처음부터 한칸씩 건너뛰면서 넣기
// 마지막으로 지금거랑 다음거랑 글자 같은지 체크하는 과정만 넣자
typedef pair<int, char> pic;

class Solution {
public: string reorganizeString(string S) {
    map<char, int> dict;
    for (int i = 0; i < S.size(); i++) {
        dict[S[i]]++;
    }

    priority_queue<pic> Q;
    for (auto kv : dict) {
        //cout << kv.first << kv.second << endl;
        Q.push({ kv.second, kv.first });
    }


    int i = 0;
    while (!Q.empty()) {
        char c = Q.top().second;
        int cnt = Q.top().first;
        //cout << c << cnt << endl;
        Q.pop();
        while (cnt--) {
            S[i] = c;
            i += 2;
            if (i >= S.size()) {
                i = 1;
            }
        }
    }

    char prev = S[0];
    for (int i = 1; i < S.size(); i++) {
        if (prev == S[i])
            return "";
        else
            prev = S[i];
    }
    return S;
}
};
```

### Integer Break
https://leetcode.com/problems/integer-break/

O(n) 솔루션이 이건가?

응 아냐 O(n^2)이야~
```c++
// product는 항상 곱이다.
// 둘로 쪼개거나, 더 쪼개거나.
// ib(n) = max([ib(i) * ib(n-i) for i in range(1,n-1)])
// ib(1) = 1, ib(2) = 1.


class Solution {
public:
    int integerBreak(int n) {
        vector<int> dp(n+1,1);

        for(int i=2;i<=n;i++){
            for(int j=1;j<i;j++){
                dp[i] = max(dp[i], max(dp[i-j]*j, (i-j)*j));
            }
        }
        return dp[n];
    }
};
```

### Find Minimum in Rotated Sorted Array II
https://leetcode.com/problems/find-minimum-in-rotated-sorted-array-ii/

얘는 O(n) 나오는거 아님?
worst가 O(n)이니까 그냥 때리셈

```c++
class Solution {
public:
    int findMin(vector<int>& nums) {
        int tmp = nums[0];
        for(auto n: nums){
            if (tmp>n)
                return n;
        }
        return tmp;
    }
};
```
